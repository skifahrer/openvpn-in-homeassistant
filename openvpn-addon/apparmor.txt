#include <tunables/global>

profile openvpn_manager flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  capability net_admin,
  capability setuid,
  capability setgid,
  capability sys_admin,
  capability dac_override,

  # Network access
  network inet stream,
  network inet6 stream,
  network inet dgram,
  network inet6 dgram,
  network unix stream,

  # Allow OpenVPN execution
  /usr/sbin/openvpn ix,

  # Allow Python execution
  /usr/bin/python3 ix,

  # Allow shell access
  /bin/bash ix,
  /usr/bin/jq ix,

  # Data directory access
  /data/ r,
  /data/** rw,

  # Configuration access
  /opt/openvpn_manager/ r,
  /opt/openvpn_manager/** r,

  # System access
  /proc/ r,
  /proc/** r,
  /sys/class/net/ r,
  /sys/class/net/** r,
  /dev/net/tun rw,
  /run/ r,
  /run/** rw,

  # Logging
  /var/log/ r,
  /var/log/** rw,

  # Required for network operations
  @{PROC}/sys/net/** r,
  /etc/protocols r,
  /etc/services r,
  /etc/resolv.conf r,
  /etc/nsswitch.conf r,
  /etc/hosts r,
  /etc/ssl/certs/ r,
  /etc/ssl/certs/** r,
}
